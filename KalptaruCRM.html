var _auth = DriveApp.getRootFolder();
const SS = SpreadsheetApp.getActiveSpreadsheet();

function authorize() {
  DriveApp.getRootFolder();
  SpreadsheetApp.getActiveSpreadsheet();
  Logger.log('Authorization complete!');
}

function doPost(e) {
  try {
    const d = JSON.parse(e.postData.contents);
    return handle(d);
  } catch(err) {
    return out({ok:false, error:err.toString()});
  }
}

function doGet(e) {
  try {
    const payload = e.parameter && e.parameter.payload;
    if (payload) {
      return handle(JSON.parse(decodeURIComponent(payload)));
    }
    return out({ok:true, status:'KalptaruCRM running'});
  } catch(err) {
    return out({ok:false, error:err.toString()});
  }
}

function handle(d) {
  if (d.action === 'save') {
    const sh = SS.getSheetByName(d.tab) || SS.insertSheet(d.tab);
    // Always keep header row up to date with current schema
    if (sh.getLastRow() === 0) {
      sh.appendRow(d.headers);
    } else {
      sh.getRange(1, 1, 1, d.headers.length).setValues([d.headers]);
    }
    // Match key: Customers → FVR ID (col 2, never blank), others → col 1
    const keyCol = (d.tab === 'Customers') ? 2 : 1;
    const keyVal = (d.tab === 'Customers') ? d.row[1] : d.row[0]; // FVR is index 1
    const lastRow = sh.getLastRow();
    let found = -1;
    if (lastRow > 1 && keyVal) {
      const colVals = sh.getRange(2, keyCol, lastRow - 1, 1).getValues().flat();
      const idx = colVals.findIndex(v => String(v).trim().toUpperCase() === String(keyVal).trim().toUpperCase());
      if (idx >= 0) found = idx + 2; // +2: 1-indexed + skip header
    }
    if (found > 0) sh.getRange(found, 1, 1, d.row.length).setValues([d.row]);
    else sh.appendRow(d.row);
    return out({ok:true});
  }
  if (d.action === 'clearSheet') {
    // Used by Push All — clears data rows but keeps sheet, then re-inserts header
    const sh = SS.getSheetByName(d.tab) || SS.insertSheet(d.tab);
    sh.clearContents();
    sh.appendRow(d.headers);
    return out({ok:true});
  }
  if (d.action === 'delete') {
    const sh = SS.getSheetByName(d.tab);
    if (sh && sh.getLastRow() > 1) {
      const keyCol = (d.tab === 'Customers') ? 2 : 1;
      const colVals = sh.getRange(2, keyCol, sh.getLastRow() - 1, 1).getValues().flat();
      const idx = colVals.findIndex(v => String(v).trim().toUpperCase() === String(d.id).trim().toUpperCase());
      if (idx >= 0) sh.deleteRow(idx + 2); // +2: 1-indexed + skip header
    }
    return out({ok:true});
  }
  if (d.action === 'load') {
    const sh = SS.getSheetByName(d.tab);
    if (!sh || sh.getLastRow() < 2) return out({rows:[]});
    return out({rows: sh.getRange(2,1,sh.getLastRow()-1,sh.getLastColumn()).getValues()});
  }
  if (d.action === 'uploadFile') {
    try {
      const folder = getOrCreateFolder('KalptaruCRM_Files');
      const blob = Utilities.newBlob(Utilities.base64Decode(d.data), d.mimeType, d.filename);
      const file = folder.createFile(blob);
      file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW);
      return out({ok:true, url:'https://drive.google.com/file/d/'+file.getId()+'/view'});
    } catch(err) {
      return out({ok:false, error:err.toString()});
    }
  }
  return out({ok:true});
}

function getOrCreateFolder(name) {
  const folders = DriveApp.getFoldersByName(name);
  return folders.hasNext() ? folders.next() : DriveApp.createFolder(name);
}

function out(data) {
  return ContentService
    .createTextOutput(JSON.stringify(data))
    .setMimeType(ContentService.MimeType.JSON);
}
