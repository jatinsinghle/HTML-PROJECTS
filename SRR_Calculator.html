// SRR Calculator — Apps Script v2 function doPost(e) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var d = JSON.parse(e.postData.contents); if (d.action === 'saveRates') { var rs = ss.getSheetByName('Rates') || ss.insertSheet('Rates'); rs.clearContents(); rs.getRange(1,1).setValue(JSON.stringify(d.rates)); return ContentService.createTextOutput('ok') .setMimeType(ContentService.MimeType.TEXT); } if (d.action === 'getRates') { var rs = ss.getSheetByName('Rates'); if (!rs || rs.getLastRow()===0) return ContentService.createTextOutput('{}') .setMimeType(ContentService.MimeType.TEXT); return ContentService.createTextOutput(rs.getRange(1,1).getValue()) .setMimeType(ContentService.MimeType.TEXT); } // Default: save entry var es = ss.getSheetByName('Entries') || ss.insertSheet('Entries'); if (es.getLastRow()===0) { es.appendRow(['Date','Type','FVR','Name','Mobile', 'Cap kW','SL kW','Final','Actual','NC Fee', 'Sanction','Stamp','Processing','Other Exp', 'Discount','SRR Incentive']); } es.appendRow([d.ts,d.type,d.fvr,d.name,d.mobile, d.kw,d.sankw,d.finalAmt,d.actualAmt,d.ncfee, d.sanction,d.stamp,d.proc,d.other,d.discount,d.srr]); return ContentService.createTextOutput('ok') .setMimeType(ContentService.MimeType.TEXT); }
